PG3DPOLYEXT

Describes a callback polygon.

    typedef struct PG3DPOLYEXT {
      PGPOLYEXT           d2;
      double              zmin;
      double              zmax;
      double              plana;
      double              planb;
      double              planc;
      double              pland;
      GetPG3DEdgeProcExt  getEdge;
      } PG3DPOLYEXT;

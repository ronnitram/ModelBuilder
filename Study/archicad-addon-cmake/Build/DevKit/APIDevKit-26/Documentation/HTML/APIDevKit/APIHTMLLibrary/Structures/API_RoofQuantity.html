API_RoofQuantity

Gives the different calculated values for roofs.

        double     bottomSurface;

        double     contourArea;

        double     ridgesLength;

        double     gablesLength;

        double     peaksLength;
        double     sideWallConLength;

        double     openingSurf;
        Int32      numOfHoles;
        Int32      numOfSkylight;
        Int32      filler_1;
        Int32      filler_2;
    } API_RoofQuantity;

Polygon::GetNormalVectorByVertex


Get normal vector by vertex.


ModelerAPI::Vector  GetNormalVectorByVertex (
    Int32 polygonVertexIndex
) const;

Return value

The normal vector for given vertex.

Description

Returns the normal Vector of this polygon at given vertex. Here is an example how to enumerate polygons and their vertexes whith normal vectors:


// body: The body, which is enumerated.

for (Int32 polygonIndex = 1; polygonIndex <= body.GetPolygonNum (); polygonIndex++) {
    ModelerAPI::Polygon polygon;
    body.GetPolygon (polygonIndex, &polygon);
    ModelerAPI::Vector polygonNormal;
    body.GetVector (polygon.GetNormalVectorIndex (), &polygonNormal);
    // Begin of polygon.
    Int32 vertexIndex = 1;
    while (vertexIndex <= polygon.GetEdgeNum ()) {
        // Begin of contour.
        for ( ; vertexIndex <= polygon.GetEdgeNum (); vertexIndex++) {
            ModelerAPI::Vector vertexNormal = polygon.GetNormalVectorByVertex (vertexIndex));
            // vertex processing
        }
        // End of contour.
        vertexIndex++;
    }
    // End of polygon.
}

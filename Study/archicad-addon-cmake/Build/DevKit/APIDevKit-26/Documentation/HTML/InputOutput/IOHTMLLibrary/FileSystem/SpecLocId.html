FileSystem::SpecLocId

enum SpecLocId {    // identifiers for special locations
    ApplicationFile,        // the running application
    ApplicationFolder,      // parent folder of the running application
    CommonDocuments,        // common documents of all users
    CommonLocalSettings,    // common local settings of all users
    CurrentFolder,          // current folder
    Desktop,                // desktop folder
    SharedComponents,       // shared components
    System,                 // system folder
    TemporaryFolder,        // folder for the temporary data
    UserDocuments,          // document folder of the current user
    UserHome,               // home folder of the current user
    UserLocalCache,         // local application cache of the current user
    UserLocalData,          // local application data of the current user
    UserLocalSettings,      // local application settings of the current user
    UserRoamingSettings     // roaming settings of the current user
};

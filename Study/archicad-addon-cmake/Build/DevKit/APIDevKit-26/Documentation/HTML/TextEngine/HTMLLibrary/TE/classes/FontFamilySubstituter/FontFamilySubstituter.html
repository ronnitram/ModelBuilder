Description

If you use a non-existing font family, then TextEngine substitute font family. TextEngine create a new virtual font family and other font family use.

Example

// ================================
class MySubstituter  : public TE::FontFamilySubstituter {
public:
  MySubstituter () {}
  ~MySubstituter () {}

  virtual const GS::UniString&  Substitute (const GS::UniString& name, GSCharCode charCode = CC_Default)
  {
    return name;  // not substitute
  }
};

// ================================
class MyFontRegistryObserver : public TE::FontRegistryObserver {
public:

  virtual void  FontAdded (const TE::FontFamily& fontFamily)
  {
    printf ("Create new fontfamily: %s - %s\n", family1.GetLocalName ().ToCStr (), family1.GetActualLocalName ().ToCStr ());
  }
};

// ================================

MySubstituter           mySubstituter;  // create substituter object
MyFontRegistryObserver  myFontRegistryObserver;

TE::textEngine.Attach (myFontRegistryObserver);
TE::textEngine.AddFontFamilySubstituter (&mySubstituter);

TE::FontFamily  family1 ("Arial");
TE::FontFamily  family2 ("example1");
TE::FontFamily  family3 ("example2", CC_Japanese);
TE::FontFamily  family4 ("example3", CC_EastEuropean);
TE::FontFamily  family5 ("Arial Symbol");

TE::textEngine.Detach (myFontRegistryObserver);
TE::textEngine.DeleteFontFamilySubstituter (&mySubstituter);

printf ("%s - %s\n", family1.GetLocalName ().ToCStr (), family1.GetActualLocalName ().ToCStr ());
printf ("%s - %s\n", family2.GetLocalName ().ToCStr (), family2.GetActualLocalName ().ToCStr ());
printf ("%s - %s\n", family3.GetLocalName ().ToCStr (), family3.GetActualLocalName ().ToCStr ());
printf ("%s - %s\n", family4.GetLocalName ().ToCStr (), family4.GetActualLocalName ().ToCStr ());
printf ("%s - %s\n", family5.GetLocalName ().ToCStr (), family5.GetActualLocalName ().ToCStr ());

// ~MySubstituter, ~MyFontRegistryObserver

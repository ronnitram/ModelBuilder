DragSourceEventProperty::SetLocationData

Sets file drag and drop data. One or more IO::Location(s). 

void SetLocationData (
    IO::Location *      locArray,
    ULong               nrOfLocations
);

Parameters

locArray
Array of IO::Locations, allocated with placement new (reference in C++ manual).
nrOfLocations
Number of allocated IO::Locations in the locations array.

Description

The location array must be allocated by placement new operator, because this drag and drop implementation will explicitly call the locations destructor.

Example

// Method Example:
void	TextControl::RichEditDragStarted (const DG::RichEditDragSourceEvent& ev, UShort* /*effect*/)
{
    if (ev.GetSource () == controlledView->editControl) {
        DG::DragSourceEventProperty*    dsep = ev.GetDragData ();
        dsep->SetData ('TDDE', &sourceDocument, sizeof (TextDocument*), DG::DragDrop::Module);
        const GS::UniString                docName = sourceDocument->GetDocumentName ();
        if (dragFileExists)
            IO::fileSystem.Delete (fileName);    // deleting the previous drag file, if there is one
        GenerateTextDragFileName (docName, &fileName);
        WriteSelectionToDragFile (fileName, controlledView->editControl);
        dragFileExists = true;
        dsep->SetLocationData (&fileName, 1);
    }
}
